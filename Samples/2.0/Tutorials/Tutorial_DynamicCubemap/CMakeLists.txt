macro( add_recursive dir retVal )
	file( GLOB_RECURSE ${retVal} ${dir}/*.h ${dir}/*.cpp ${dir}/*.c )
endmacro()

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/include)

include_directories(${CMAKE_SOURCE_DIR}/Components/Hlms/Common/include)
ogre_add_component_include_dir(Hlms/Pbs)

add_recursive( ./ SOURCE_FILES )

ogre_add_executable(Sample_Tutorial_DynamicCubemap WIN32 MACOSX_BUNDLE ${SOURCE_FILES} ${SAMPLE_COMMON_RESOURCES} ${SAMPLE_COMMON_COMPUTE_RESOURCES})

target_link_libraries(Sample_Tutorial_DynamicCubemap ${OGRE_LIBRARIES} ${OGRE_SAMPLES_LIBRARIES})
ogre_config_sample_lib(Sample_Tutorial_DynamicCubemap)
ogre_config_sample_pkg(Sample_Tutorial_DynamicCubemap)

if( APPLE )
	set( Sample_RESOURCES ${SAMPLE_COMMON_RESOURCES} ${SAMPLE_COMMON_COMPUTE_RESOURCES} )
	set_target_properties( Sample_Tutorial_DynamicCubemap PROPERTIES RESOURCE "${Sample_RESOURCES}" )
	set_target_properties( Sample_Tutorial_DynamicCubemap PROPERTIES MACOSX_BUNDLE_INFO_PLIST ${SAMPLES_PLIST} )
endif()
